🚀 基础用法

import asyncio
from penshot import PenshotFunction

async def main():
    # 创建智能体
    agent = PenshotFunction()

    # 提交任务并等待结果
    result = agent.breakdown_script("""
        场景：现代办公室
        人物：小李（程序员）
        动作：小李正在写代码，突然接到电话
    """)

    if result.success:
        print(f"生成成功！共 {len(result.data['shots'])} 个镜头")
    else:
        print(f"生成失败: {result.error}")

asyncio.run(main())

🌐 REST API

# 启动服务
penshot serve-rest --port 8000

# 提交任务
curl -X POST http://localhost:8000/api/v1/storyboard \
  -H "Content-Type: application/json" \
  -d '{"script": "你的剧本内容...", "language": "zh"}'

# 查询状态
curl http://localhost:8000/api/v1/status/{task_id}

# 获取结果
curl http://localhost:8000/api/v1/result/{task_id}
